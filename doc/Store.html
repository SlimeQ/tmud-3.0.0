  def stats
    rooms = objs = scripts = characters = accounts = 0
    self.each do |val|
      case val
      when Room
        rooms += 1
      when Character
        characters += 1
      when GameObject
        objs += 1
      when Script
        scripts += 1
      when Account
        accounts += 1
      end
    end
    stats="[COLOR Cyan]
---* Database Statistics *---
  Rooms      - #{rooms}
  Objects    - #{objs}
  Scripts    - #{scripts}
  Accounts   - #{accounts}
  Characters - #{characters}
  Total Objects - #{rooms+objs+characters+accounts+scripts}
  Highest OID in use - #{@dbtop}
---*                     *---
[/COLOR]
"
  end
