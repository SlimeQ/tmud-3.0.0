def is_match? str
  return false if self.empty? || str.nil? || str.empty?
  lst = self.split(' ')
  lst.each do |s|
    return true if str.downcase == s.slice(0...str.size).downcase
  end
  false
end

    # File lib/utility/utility.rb, line 53
def proper_name
  str = self.dup
  str.gsub!(%r\d+/,'')
  str.gsub!(%r\w+/) {|m|  m.downcase!; m[0] = m[0].chr.upcase; m}
  str
end
