def write(msg)
  @outbuffer << msg
  n = @sock.send(@outbuffer, 0)
  # save unsent data for next call
  @outbuffer.slice!(0...n)
  @outbuffer.size == 0
rescue Exception
  @outbuffer = ""  # Does it really matter?
  raise
end
